```python
"""
AgentUI 前端 E2E 测试
使用 Playwright 验证前端页面核心功能
"""
import asyncio
from playwright.async_api import async_playwright

async def run_tests():
    results = []  # 记录测试结果
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()

        # 收集 console error
        console_errors = []
        page.on("console", lambda msg: console_errors.append(msg.text) if msg.type == "error" else None)

        # 3.1 打开页面
        await page.goto("http://localhost:5173", wait_until="networkidle", timeout=15000)
        results.append(("页面加载", True, "页面成功加载"))

        # 3.2 页面标题检查
        title = await page.title()
        title_ok = "AgentUI" in title or "agentui" in title.lower()
        results.append(("页面标题包含 AgentUI", title_ok, f"实际标题: {title}"))

        # 3.3 StatusBar 显示 "小舒舒办公室"
        status_bar = await page.query_selector("text=小舒舒办公室")
        results.append(("StatusBar 显示 '小舒舒办公室'", status_bar is not None, "StatusBar 文本检查"))

        # 3.4 AgentCard 数量检查
        # 等待 AgentCard 渲染完毕
        await page.wait_for_timeout(2000)
        cards = await page.query_selector_all(".agent-card")
        if len(cards) == 0:
            # 备用选择器: 尝试其他可能的类名
            cards = await page.query_selector_all("[class*='agent']")
        cards_ok = len(cards) >= 9
        results.append(("9 个 AgentCard 显示", cards_ok, f"实际找到 {len(cards)} 个卡片"))

        # 3.5 PixiJS canvas 元素存在
        canvas = await page.query_selector("canvas")
        results.append(("PixiJS canvas 元素存在", canvas is not None, "canvas 标签检查"))

        # 3.6 无 console error
        no_errors = len(console_errors) == 0
        results.append(("无 console error", no_errors, f"错误数: {len(console_errors)}" + (f", 内容: {console_errors[:3]}" if console_errors else "")))

        await browser.close()

    # 输出结果
    print("\n" + "=" * 60)
    print("AgentUI 前端 E2E 测试报告")
    print("=" * 60)
    all_pass = True
    for name, passed, detail in results:
        status = "PASS" if passed else "FAIL"
        if not passed:
            all_pass = False
        print(f"  [{status}] {name} — {detail}")
    print("=" * 60)
    print(f"总计: {len(results)} 项, 通过: {sum(1 for _,p,_ in results if p)} 项, 失败: {sum(1 for _,p,_ in results if not p)} 项")
    if all_pass:
        print("结论: 全部通过")
    else:
        print("结论: 存在失败项，请检查")
    print("=" * 60)

asyncio.run(run_tests())
```

### 集成测试脚本

测试脚本路径: `/Users/young/Desktop/code/AgentUI/tests/integration_test.py`

```python
"""
AgentUI 集成测试
验证前后端联动: 状态修改 -> 前端自动更新、Agent 移动
"""
import asyncio
import json
import urllib.request
from playwright.async_api import async_playwright

API_BASE = "http://localhost:19000"
FRONTEND_URL = "http://localhost:5173"

def api_post(path, data):
    """发送 POST 请求到后端"""
    req = urllib.request.Request(
        f"{API_BASE}{path}",
        data=json.dumps(data).encode(),
        headers={"Content-Type": "application/json"},
        method="POST"
    )
    with urllib.request.urlopen(req) as resp:
        return json.loads(resp.read())

def api_get(path):
    """发送 GET 请求到后端"""
    with urllib.request.urlopen(f"{API_BASE}{path}") as resp:
        return json.loads(resp.read())

async def test_state_update():
    """测试: 修改状态后前端自动更新"""
    print("\n[集成测试 4.1] 状态修改后前端自动更新")
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()
        await page.goto(FRONTEND_URL, wait_until="networkidle", timeout=15000)

        # 截图: 修改前
        await page.screenshot(path="/tmp/agentui_before_state.png")

        # 通过 API 修改状态
        api_post("/set_state", {"state": "meeting"})
        print("  已通过 API 设置状态为 meeting")

        # 等待前端轮询/推送更新
        await page.wait_for_timeout(3000)

        # 截图: 修改后
        await page.screenshot(path="/tmp/agentui_after_state.png")

        # 验证: 检查页面内容是否包含新状态
        content = await page.content()
        has_update = "meeting" in content.lower()
        print(f"  页面是否包含 'meeting': {has_update}")
        print(f"  [{'PASS' if has_update else 'FAIL'}] 状态修改后前端自动更新")

        await browser.close()
    return has_update

async def test_agent_movement():
    """测试: Agent 状态切换后小人移动 (canvas 截图对比)"""
    print("\n[集成测试 4.2] Agent 状态切换后小人移动")
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()
        await page.goto(FRONTEND_URL, wait_until="networkidle", timeout=15000)
        await page.wait_for_timeout(2000)

        # 截图 canvas: 切换前
        canvas = await page.query_selector("canvas")
        if not canvas:
            print("  [FAIL] 未找到 canvas 元素")
            await browser.close()
            return False

        screenshot_before = await canvas.screenshot()
        print("  已截取 canvas 切换前状态")

        # 通过 API 切换状态，触发小人移动
        api_post("/set_state", {"state": "working"})
        print("  已通过 API 设置状态为 working")

        # 等待动画完成
        await page.wait_for_timeout(5000)

        # 截图 canvas: 切换后
        screenshot_after = await canvas.screenshot()
        print("  已截取 canvas 切换后状态")

        # 对比两张截图的字节内容
        changed = screenshot_before != screenshot_after
        print(f"  Canvas 内容是否变化: {changed}")
        print(f"  [{'PASS' if changed else 'FAIL'}] Agent 状态切换后小人移动")

        await browser.close()
    return changed

async def main():
    print("=" * 60)
    print("AgentUI 集成测试报告")
    print("=" * 60)

    results = []
    results.append(("状态修改后前端自动更新", await test_state_update()))
    results.append(("Agent 状态切换后小人移动", await test_agent_movement()))

    print("\n" + "=" * 60)
    print("集成测试汇总")
    print("=" * 60)
    for name, passed in results:
        print(f"  [{'PASS' if passed else 'FAIL'}] {name}")
    total_pass = sum(1 for _, p in results if p)
    print(f"总计: {len(results)} 项, 通过: {total_pass} 项, 失败: {len(results) - total_pass} 项")
    print("=" * 60)

asyncio.run(main())
```
